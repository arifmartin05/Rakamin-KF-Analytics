-- -- -- -- -- from data rakamin-kf-analytics-458515.kimia_farma.kf_abc1; buat Top 5 Cabang Dengan Rating Tertinggi, namun Rating Transaksi Terendah


SELECT
  branch_name,
  AVG(rating) AS average_rating,
  MIN(rating) AS min_rating
FROM
  `rakamin-kf-analytics-458515`.`kimia_farma`.`kf_abc1`
GROUP BY
  branch_name
ORDER BY
  average_rating DESC,
  min_rating ASC
LIMIT
  5;
